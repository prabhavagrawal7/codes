\begin{algorithm}
\caption{Covariance Matching-Based 1-D Kalman Filter}
Initialize $x_0, P_0, Q, R$ \\
for $k = 1$ to $N:$ \\
\qquad    Predict equations: \\
\qquad    $\hat{s}^-_k = \hat{s}_{k-1}$ \\
\qquad    $P^-_k = P_{k-1} + Q$ \\
\qquad    Update equations: \\
\qquad    $K_k = \frac{P^-_k}{P^-_k + R}$ \\
\qquad    $\hat{s}_k = \hat{s}^-_k + K_k(z_k - \hat{s}^-_k)$ \\
\qquad    $P_k = (1 - K_k)P^-_k$ \\
\end{algorithm}
