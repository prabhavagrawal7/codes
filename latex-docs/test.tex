\section{Algorithm}
\begin{frame}

\begin{algorithm}[H]
\caption{Calculate Fibonacci sequence}
\begin{algorithmic}
\STATE Input: n (number of terms)
\STATE Output: Fibonacci sequence up to n terms
\STATE initialize a = 0, b = 1
\STATE print a
\STATE print b
\FOR{$i = 2$ to $n$}
\STATE c = a + b
\STATE print c
\STATE a = b
\STATE b = c
\ENDFOR
\end{algorithmic}
\end{algorithm}
